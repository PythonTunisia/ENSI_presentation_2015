def say_many_hellos(name, times=3):
    """
    Says Hello N times.

    @name: string
    @times: int

    """
    messages = ["hello {}\n".formt(name) for name in range(times)]
    # Joins the list of messages and returns them as one String.
    return "".join(messages)
